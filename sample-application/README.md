    useridentity.username = '[user-name-1]' OR

# QUERY 3 : 지난 7일간 사용한 user들의 list

FROM "default"."[your-table-name]"

    from_iso8601_timestamp(eventtime) > date_add('day', -7, now()) 
LIMIT 10;
